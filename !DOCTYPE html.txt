  <script>
    let catalogData = [];
    let editMode = false;

    // Load catalog from catalog.json
    async function loadCatalog() {
      try {
        const response = await fetch('catalog.json');
        catalogData = await response.json();
        renderCatalog();
      } catch (error) {
        console.error('Error loading catalog:', error);
      }
    }

    function renderCatalog() {
      const catalog = document.getElementById('catalog');
      catalog.innerHTML = '';

      catalogData.forEach((item, index) => {
        const div = document.createElement('div');
        div.className = 'product';

        if (editMode) {
          div.innerHTML = `
            <input class="edit-input" value="${item.name}" onchange="updateField(${index}, 'name', this.value)" />
            <input class="edit-input" value="${item.price}" onchange="updateField(${index}, 'price', this.value)" />
            <input class="edit-input" value="${item.image}" onchange="updateField(${index}, 'image', this.value)" />
            <textarea class="edit-input" onchange="updateField(${index}, 'description', this.value)">${item.description}</textarea>
          `;
        } else {
          div.innerHTML = `
            <img src="${item.image}" alt="${item.name}">
            <h3>${item.name}</h3>
            <p class="price">${item.price}</p>
            <p>${item.description}</p>
          `;
        }
        catalog.appendChild(div);
      });
    }

    function updateField(index, field, value) {
      catalogData[index][field] = value;
    }

    function toggleEdit(enable) {
      editMode = enable;
      renderCatalog();
      document.getElementById('btnToggleEdit').textContent = editMode ? 'Exit Edit' : 'Edit Catalog';
    }

    // Simple admin mode with password
    const ADMIN_PASSWORD = '31marche2025';
    const ADMIN_SESSION_KEY = '31marche_is_admin';

    function isAdminSession() {
      return sessionStorage.getItem(ADMIN_SESSION_KEY) === '1';
    }

    function promptAdminPassword() {
      const input = prompt('Enter admin password to edit:');
      if (input === ADMIN_PASSWORD) {
        sessionStorage.setItem(ADMIN_SESSION_KEY, '1');
        alert('Edit mode unlocked!');
        toggleEdit(true);
      } else {
        alert('Wrong password.');
      }
    }

    const btnToggle = document.getElementById('btnToggleEdit');
    btnToggle.addEventListener('click', () => {
      if (isAdminSession()) {
        toggleEdit(!editMode);
      } else {
        promptAdminPassword();
      }
    });

    loadCatalog();
  </script>
